18  package builders.dsl.spreadsheet.query.poi;
19  
20  import builders.dsl.spreadsheet.api.*;
21  import org.apache.poi.xssf.usermodel.XSSFCellStyle;
22  import org.apache.poi.xssf.usermodel.extensions.XSSFCellBorder;
23  
24  class PoiCellStyle implements CellStyle {
25      PoiCellStyle(XSSFCellStyle style) {
26          this.style = style;
27      }
28  
29      @Override
30      public Color getForeground() {
31          if (style.getFillForegroundColorColor() == null) {
32              return null;
33          }
34          if (style.getFillForegroundColorColor().getRGB() == null) {
35              return null;
36          }
37          return new Color(style.getFillForegroundColorColor().getRGB());
38      }
39  
40      @Override
41      public Color getBackground() {
42          if (style.getFillBackgroundColorColor() == null) {
43              return null;
44          }
45          if (style.getFillBackgroundColorColor().getRGB() == null) {
46              return null;
47          }
48          return new Color(style.getFillBackgroundColorColor().getRGB());
49      }
50  
51      @Override
52      public ForegroundFill getFill() {
53          switch (style.getFillPattern()) {
54              case NO_FILL:
55                  return ForegroundFill.NO_FILL;
56              case SOLID_FOREGROUND:
57                  return ForegroundFill.SOLID_FOREGROUND;
58              case FINE_DOTS:
59                  return ForegroundFill.FINE_DOTS;
60              case ALT_BARS:
61                  return ForegroundFill.ALT_BARS;
62              case SPARSE_DOTS:
63                  return ForegroundFill.SPARSE_DOTS;
64              case THICK_HORZ_BANDS:
65                  return ForegroundFill.THICK_HORZ_BANDS;
66              case THICK_VERT_BANDS:
67                  return ForegroundFill.THICK_VERT_BANDS;
68              case THICK_BACKWARD_DIAG:
69                  return ForegroundFill.THICK_BACKWARD_DIAG;
70              case THICK_FORWARD_DIAG:
71                  return ForegroundFill.THICK_FORWARD_DIAG;
72              case BIG_SPOTS:
73                  return ForegroundFill.BIG_SPOTS;
74              case BRICKS:
75                  return ForegroundFill.BRICKS;
76              case THIN_HORZ_BANDS:
77                  return ForegroundFill.THIN_HORZ_BANDS;
78              case THIN_VERT_BANDS:
79                  return ForegroundFill.THIN_VERT_BANDS;
80              case THIN_BACKWARD_DIAG:
81                  return ForegroundFill.THIN_BACKWARD_DIAG;
82              case THIN_FORWARD_DIAG:
83                  return ForegroundFill.THIN_FORWARD_DIAG;
84              case SQUARES:
85                  return ForegroundFill.SQUARES;
86              case DIAMONDS:
87                  return ForegroundFill.DIAMONDS;
88          }
89          return null;
90      }
91  
92      @Override
93      public int getIndent() {
94          return style.getIndention();
95      }
96  
97      @Override
98      public int getRotation() {
99          return style.getRotation();
100     }
101 
102     @Override
103     public String getFormat() {
104         return style.getDataFormatString();
105     }
106 
107     @Override
108     public Font getFont() {
109         return style.getFont() != null ? new PoiFont(style.getFont()) : null;
110     }
111 
112     @Override
113     public Border getBorder(Keywords.BorderSide borderSide) {
114         if (Keywords.BorderSide.TOP.equals(borderSide)) {
115             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.TOP), style.getBorderTopEnum());
116         } else if (Keywords.BorderSide.BOTTOM.equals(borderSide)) {
117             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.BOTTOM), style.getBorderBottomEnum());
118         } else if (Keywords.BorderSide.LEFT.equals(borderSide)) {
119             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.LEFT), style.getBorderLeftEnum());
120         } else if (Keywords.BorderSide.RIGHT.equals(borderSide)) {
121             return new PoiBorder(style.getBorderColor(XSSFCellBorder.BorderSide.RIGHT), style.getBorderRightEnum());
122         }
123         return new PoiBorder(null, null);
124     }
125 
126     private final XSSFCellStyle style;
127 }
